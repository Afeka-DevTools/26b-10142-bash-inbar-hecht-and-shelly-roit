git clone [https://github.com/](https://github.com/)[username]/[repository-name].git

# 2. כניסה לתיקיית הפרויקט שנוצרה
cd [Bash_Scripts]
# 3. בשביל לבדוק אם יש לקבוץ הרשאות תריצו את הפקודה
ls -l script_name.sh

# 4. מתן הרשאות ריצה לכל קובצי הסקריפט
chmod +x *.sh

./script_name.sh
